\begin{frame}
  \frametitle{Recall und Precision}
  Recall:

  \emph{In wie viel Prozent der Fälle wurde der Nutzer richtig erkannt?}

  \begin{displaymath}
    R = \frac{TP}{TP + FN}
  \end{displaymath}  

  \hbox{}
  Precision:

  \emph{In wie viel Prozent der Fälle in denen der Algorithmus gedacht hat es handle sich um den Nutzer war dies auch wirklich der Fall?}

  \begin{displaymath}
    P = \frac{TP}{TP + FP}    
  \end{displaymath}
\end{frame}

\section{Fazit}

\subsection{Bewertung des Verfahrens}
\begin{frame}
  \frametitle{Bewertung des Verfahrens}
  \begin{itemize}
  \item Im Zusammenspiel mit einer klassischen Pineingabe durchaus als Merkmal für eine Authentifizierung geeignet
  \item Resistent gegenüber \glqq{}Shoulder Surfing\grqq{}
  \item Leichter zu ersetzten als ein biometrisches Merkmal
  \item Unterschiedliche Performance für verschiedene Nutzer
    \begin{itemize}
    \item Touchscreen nicht optimal
    \item Unterschiedlich viel Erfahrung im Umgang mit Touchscreens
    \item Unterschiedlich viele Testdaten
    \end{itemize}
  \end{itemize}
\end{frame}
\subsection{Ausblick}
\begin{frame}
  \frametitle{Ausblick}
  Es gibt noch viele Möglichkeiten die Performance des Verfahrens zu verbessern:
  \begin{itemize}
  \item Umfangreicherer Feldtest unter diverseren Testbedingungen    
  \item Optimieren der Klassifikatoren
  \item Verwenden verschiedener Klassifikatoren und Abstimmung über ein Mehrheitsvotum
  \item Verwendung eines zuverlässigeren Eingabegerätes
  \end{itemize}
\end{frame}
